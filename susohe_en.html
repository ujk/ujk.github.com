require "rubygems"
require "sinatra"

mat = []
(1..81).each do
    mat << "123456789" 
end

def check mat
    #search for single values
    singles=[]
    for row in 1..9
        for col in 1..9
            v = mat[(row-1)*9+(col-1)]
            if v.length==1
                singles << (row.to_s + col.to_s + v)
            end
        end
    end
    #remove singles from cell groups
    for row in 1..9
        for col in 1..9
            if mat[(row-1)*9+(col-1)].length != 1       #if there are more than one value in a cell
                "123456789".each_char do |x|                  # for all available values 
                                        # if there is same value at the row as single 
                    if singles.detect {|v| v =~ (Regexp.new(row.to_s + "[1-9]" + x))} 
                        mat[(row-1)*9+(col-1)].delete! x        # delete this number
                    # else add that value to available values
                    else        
                        mat[(row-1)*9+(col-1)] += x        
                    end
                end
            end
        end
    end
end

get "/" do
    haml :index, :locals=>{:mat => mat}
end

get "/set/:x/:xx" do
    x = params[:x].to_i
    xx = params[:xx]
    mat[x-1]=xx
    check mat
    redirect("/")
end
  
get "/reset/:x" do
    x = params[:x].to_i
    mat[x-1]="123456789"
    check mat
    redirect "/"
end

First we produce a list of cells which include a single value. This list has string elements. This elements are 3 character long, first one is row number, second one is column number and last one is cell's value. For example if 5th row's 3th column has value of 4, the array element of this cell will be "534".

After create singles array, we check all cells to delete values if there is single value in same row. If tested value isn't in singles array at same row, we add this value to cell's values. With "singles.detect {|v| v =~ (Regexp.new(row.to_s + "[1-9]" + x))}"  syntax we hunt same values on the row in singles array.
